534

883  entry_in.key = pl->key;
884  entry_in.data = pl;
885  hsearch_r( entry_in, ENTER, &entry_outp, &lsredis_preset_ht);
886  }
887  lslogging_log_message( "lsredis_load_presets: done increasing preset hash table size.", lsredis_preset_max_n);
888  }
889  }
890  free( preset_name);
891  }
