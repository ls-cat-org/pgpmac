852  //

923  //

925  lspmac_SockSendDPline( "Exposure", "&1 P170=%.1f P171=%.1f P173=%.1f P174=0 P175=%.1f P176=0 P177=1 P178=0 P180=%.1f M431=1 &1B131R",
926  p170, p171, p173, p175, p180);
927 
928  //
929  // We could look for the "Exposure command accepted" event at this point.
930  //
931  //
932  // wait for the shutter to open
933  //
934  clock_gettime( CLOCK_REALTIME, &now);
935  timeout.tv_sec = now.tv_sec + 10;
936  timeout.tv_nsec = now.tv_nsec;
937 
938  err = 0;
939  pthread_mutex_lock( &lspmac_shutter_mutex);
940  while( err == 0 && lspmac_shutter_has_opened == 0)
941  err = pthread_cond_timedwait( &lspmac_shutter_cond, &lspmac_shutter_mutex, &timeout);
942 
943  if( err == ETIMEDOUT) {
944  pthread_mutex_unlock( &lspmac_shutter_mutex);
945  lslogging_log_message( "md2cmds_collect: Timed out waiting for shutter to open. Data collection aborted.");
946  lsevents_send_event( "Data Collection Aborted");
947  return;
948  }
949

951  //
952  // wait for the shutter to close
953  //
954  clock_gettime( CLOCK_REALTIME, &now);
955  timeout.tv_sec = now.tv_sec + 4 + exp_time; // hopefully 4 seconds is long enough to never catch a legitimate shutter close and short enough to bail when something is really wrong
956  timeout.tv_nsec = now.tv_nsec;
957 
958  err = 0;
959  while( err == 0 && lspmac_shutter_state == 1)
960  err = pthread_cond_timedwait( &lspmac_shutter_cond, &lspmac_shutter_mutex, &timeout);
961  pthread_mutex_unlock( &lspmac_shutter_mutex);
962 
963 
964  if( err == ETIMEDOUT) {
965  pthread_mutex_unlock( &lspmac_shutter_mutex);
966  lslogging_log_message( "md2cmds_collect: Timed out waiting for shutter to close. Data collection aborted.");
967  lsevents_send_event( "Data Collection Aborted");
968  return;
969  }
970 
971 
972  //
973  // Signal the detector to start reading out
974  //
975  lspg_query_push( NULL, "SELECT px.unlock_diffractometer()");
976 
977  //
978  // Update the shot status
979  //
980  lspg_query_push( NULL, "SELECT px.shots_set_state(%lld, 'Writing')", skey);
981 
982  //
983  // reset shutter has opened flag
984  //
985  lspmac_SockSendDPline( NULL, "P3001=0");
986 
987  //
988  // Wait for omega to stop moving
989  //
990  if( md2cmds_move_wait( 10.0)) {
991  lslogging_log_message( "md2cmds_collect: Giving up waiting for omega to stop moving. Data collection aborted.");
992  lsevents_send_event( "Data Colection Aborted");
993  return;
994  }
995 
996  //
997  // Move the center/alignment stages to the next position
998  //
999  // TODO: position omega for the next shot. During data collection the motion program
1000  // makes a good guess but for ortho snaps it is wrong. We should add an argument to the motion program
1001  //
1002 
1003 
1004  if( !lspg_nextshot.active2_isnull && lspg_nextshot.active2) {
1005  if(
1006  (fabs( lspg_nextshot.cx2 - cenx->position) > 0.1) ||
1007  (fabs( lspg_nextshot.cy2 - ceny->position) > 0.1) ||
1008  (fabs( lspg_nextshot.ax2 - alignx->position) > 0.1) ||
1009  (fabs( lspg_nextshot.ay2 - aligny->position) > 0.1) ||
1010  (fabs( lspg_nextshot.az2 - alignz->position) > 0.1)) {
1011 
1012  center_request = 1;
1013  md2cmds_move_prep();
1014  md2cmds_mvcenter_move( lspg_nextshot.cx, lspg_nextshot.cy, lspg_nextshot.ax, lspg_nextshot.ay, lspg_nextshot.az);
1015  }
1016  }
1017  }
