132  //

134  // Then sets them.
135  // Each time we wait until we've read back
136  // the changed values
137  //
138  // This guarantees that when we are waiting for motion to stop that it did, in fact, start
139  //

143  //

145 
146  //
147  // Make sure it propagates
148  //
149  pthread_mutex_lock( &lspmac_moving_mutex);
150  while( lspmac_moving_flags & 6)
151  pthread_cond_wait( &lspmac_moving_cond, &lspmac_moving_mutex);
152  pthread_mutex_unlock( &lspmac_moving_mutex);
153 
154  //
155  // Set the centering and alignment stage flags
