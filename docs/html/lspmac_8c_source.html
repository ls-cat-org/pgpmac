764  ls_pmac_state = LS_PMAC_STATE_WACK_RR;

804  //

809  //

817  //

819  receiveBuffer[i] = 0;
820  break;
821  }
822  }
823  receiveBufferIn = i;
824  }
825

827

829  case LS_PMAC_STATE_WACK_NFR:
830  receiveBuffer[--receiveBufferIn] = 0;
831  cmd = lspmac_pop_reply();
832  ls_pmac_state = LS_PMAC_STATE_IDLE;
833  break;
834  case LS_PMAC_STATE_WACK:

836  ls_pmac_state = LS_PMAC_STATE_RR;
837  break;
838  case LS_PMAC_STATE_WACK_CC:
839  receiveBuffer[--receiveBufferIn] = 0;
840  ls_pmac_state = LS_PMAC_STATE_CR;
841  break;
842  case LS_PMAC_STATE_WACK_RR:
843  receiveBufferIn -= 2;
844  if( receiveBuffer[receiveBufferIn])
845  ls_pmac_state = LS_PMAC_STATE_GB;
846  else
847  ls_pmac_state = LS_PMAC_STATE_RR;
848  receiveBuffer[receiveBufferIn] = 0;
849  break;
850  case LS_PMAC_STATE_GMR:
851  cmd = lspmac_pop_reply();
852  ls_pmac_state = LS_PMAC_STATE_IDLE;
853  break;
854 
855  case LS_PMAC_STATE_WCR:
856  cmd = lspmac_pop_reply();
857  ls_pmac_state = LS_PMAC_STATE_IDLE;
858  break;
859  case LS_PMAC_STATE_WGB:
860  if( foundEOCR) {
861  cmd = lspmac_pop_reply();
862  ls_pmac_state = LS_PMAC_STATE_IDLE;
863  } else {
864  ls_pmac_state = LS_PMAC_STATE_RR;
865  }
866  break;
867  }

871  cmd->onResponse( cmd, receiveBufferIn, receiveBuffer);
872  receiveBufferIn = 0;
873  }
874  }
