808 void lspmac_Service(
809  struct pollfd *evt
810  ) {

812  static int receiveBufferSize = 0; // size of receiveBuffer
813  static int receiveBufferIn = 0; // next location to write to in receiveBuffer
814  pmac_cmd_queue_t *cmd; // maybe the command we are servicing
815  ssize_t nsent, nread; // nbytes dealt with
816  int i; // loop counter
817  int foundEOCR; // end of command response flag
818 
819  if( evt->revents & (POLLERR | POLLHUP | POLLNVAL)) {
820  if( evt->fd != -1) {
821  close( evt->fd);
822  evt->fd = -1;
823  }
824  ls_pmac_state = LS_PMAC_STATE_DETACHED;
825  return;
826  }
827 
828 
829  if( evt->revents & POLLOUT) {
830 
831  switch( ls_pmac_state) {
832  case LS_PMAC_STATE_DETACHED:
833  break;
834  case LS_PMAC_STATE_IDLE:
835  break;
836 
837  case LS_PMAC_STATE_SC:
838  cmd = lspmac_pop_queue();
839  if( cmd == NULL)
840  return;
841 
842  if( cmd->pcmd.Request == VR_PMAC_GETMEM) {
843  nsent = send( evt->fd, cmd, pmac_cmd_size, 0);
844  if( nsent != pmac_cmd_size) {
845  lslogging_log_message( "Could only send %d of %d bytes....Not good.", (int)nsent, (int)(pmac_cmd_size));
846  }
847  } else {
848  nsent = send( evt->fd, cmd, pmac_cmd_size + ntohs(cmd->pcmd.wLength), 0);
849  gettimeofday( &pmac_time_sent, NULL);
850  if( nsent != pmac_cmd_size + ntohs(cmd->pcmd.wLength)) {
851  lslogging_log_message( "Could only send %d of %d bytes....Not good.", (int)nsent, (int)(pmac_cmd_size + ntohs(cmd->pcmd.wLength)));
852  }
853  }
854 
855  if( cmd->pcmd.Request == VR_PMAC_SENDCTRLCHAR)
856  ls_pmac_state = LS_PMAC_STATE_WACK_CC;
857  else if( cmd->pcmd.Request == VR_PMAC_GETMEM)
858  ls_pmac_state = LS_PMAC_STATE_GMR;
859  else if( cmd->no_reply == 0)
860  ls_pmac_state = LS_PMAC_STATE_WACK;
861  else
862  ls_pmac_state = LS_PMAC_STATE_WACK_NFR;
863  break;
864 
865  case LS_PMAC_STATE_CR:
866  nsent = send( evt->fd, &cr_cmd, pmac_cmd_size, 0);
867  gettimeofday( &pmac_time_sent, NULL);
868  ls_pmac_state = LS_PMAC_STATE_WCR;
869  break;
870 
871  case LS_PMAC_STATE_RR:
872  nsent = send( evt->fd, &rr_cmd, pmac_cmd_size, 0);
873  gettimeofday( &pmac_time_sent, NULL);
874  ls_pmac_state = LS_PMAC_STATE_WACK_RR;
875  break;
876 
877  case LS_PMAC_STATE_GB:
878  nsent = send( evt->fd, &gb_cmd, pmac_cmd_size, 0);
879  gettimeofday( &pmac_time_sent, NULL);
880  ls_pmac_state = LS_PMAC_STATE_WGB;
881  break;
882  }
883  }
884 
885  if( evt->revents & POLLIN) {
886 
887  if( receiveBufferSize - receiveBufferIn < 1400) {
888  char *newbuff;
889 
890  receiveBufferSize += 1400;
891  newbuff = calloc( receiveBufferSize, sizeof( unsigned char));
892  if( newbuff == NULL) {
893  lslogging_log_message( "lspmac_Service: Out of memory");
894  exit( -1);
895  }
896  if( receiveBuffer != NULL) {
897  memcpy( newbuff, receiveBuffer, receiveBufferIn);
898  free(receiveBuffer);
899  }
900  receiveBuffer = newbuff;
901  }
902 
903  nread = read( evt->fd, receiveBuffer + receiveBufferIn, 1400);
904 
905  foundEOCR = 0;
906  if( ls_pmac_state == LS_PMAC_STATE_GMR) {
907  //
908  // get memory returns binary stuff, don't try to parse it
909  //
910  receiveBufferIn += nread;
911  } else {
912  //
913  // other commands end in 6 if OK, 7 if not
914  //
915  for( i=receiveBufferIn; i<receiveBufferIn+nread; i++) {
916  if( receiveBuffer[i] == 7) {
917  //
918  // Error condition
919  //
920  lspmac_Error( &(receiveBuffer[i]));
921  receiveBufferIn = 0;
922  return;
923  }
924  if( receiveBuffer[i] == 6) {
925  //
926  // End of command response
927  //
928  foundEOCR = 1;
929  receiveBuffer[i] = 0;
930  break;
931  }
932  }
933  receiveBufferIn = i;
934  }
935 
936  cmd = NULL;
937 
938  switch( ls_pmac_state) {
939  case LS_PMAC_STATE_WACK_NFR:
940  receiveBuffer[--receiveBufferIn] = 0;
941  cmd = lspmac_pop_reply();
942  ls_pmac_state = LS_PMAC_STATE_IDLE;
943  break;
944  case LS_PMAC_STATE_WACK:
945  receiveBuffer[--receiveBufferIn] = 0;
946  ls_pmac_state = LS_PMAC_STATE_RR;
947  break;
948  case LS_PMAC_STATE_WACK_CC:
949  receiveBuffer[--receiveBufferIn] = 0;
950  ls_pmac_state = LS_PMAC_STATE_CR;
951  break;
952  case LS_PMAC_STATE_WACK_RR:
953  receiveBufferIn -= 2;
954  if( receiveBuffer[receiveBufferIn])
955  ls_pmac_state = LS_PMAC_STATE_GB;
956  else
957  ls_pmac_state = LS_PMAC_STATE_RR;
958  receiveBuffer[receiveBufferIn] = 0;
959  break;
960  case LS_PMAC_STATE_GMR:
961  cmd = lspmac_pop_reply();
962  ls_pmac_state = LS_PMAC_STATE_IDLE;
963  break;
964 
965  case LS_PMAC_STATE_WCR:
966  cmd = lspmac_pop_reply();
967  ls_pmac_state = LS_PMAC_STATE_IDLE;
968  break;
969  case LS_PMAC_STATE_WGB:
970  if( foundEOCR) {
971  cmd = lspmac_pop_reply();
972  ls_pmac_state = LS_PMAC_STATE_IDLE;
973  } else {
974  ls_pmac_state = LS_PMAC_STATE_RR;
975  }
976  break;
977  }
978 
979 
980  if( cmd != NULL && cmd->onResponse != NULL) {
981  cmd->onResponse( cmd, receiveBufferIn, receiveBuffer);
982  receiveBufferIn = 0;
983  }
984  }
985 }

1879  }

3426 int lspmac_moveabs_wait( lspmac_motor_t *mp, double timeout_secs) {
3427  struct timespec timeout, now;
3428  double isecs, fsecs;
3429  int err;
3430 
3431  //
3432  // Copy the queue item for the most recent move request
3433  //
3434  clock_gettime( CLOCK_REALTIME, &now);
3435 
3436  fsecs = modf( timeout_secs, &isecs);
3437 
3438  timeout.tv_sec = now.tv_sec + (long)floor( isecs);
3439  timeout.tv_nsec = now.tv_nsec + (long)floor( fsecs * 1.0e9);
3440 
3441  timeout.tv_sec += timeout.tv_nsec / 1000000000;
3442  timeout.tv_nsec %= 1000000000;
3443 
3444  err = 0;
3445  pthread_mutex_lock( &(mp->mutex));
3446 
3447  while( err == 0 && mp->command_sent == 0)
3448  err = pthread_cond_timedwait( &mp->cond, &mp->mutex, &timeout);
3449  pthread_mutex_unlock( &(mp->mutex));
3450  if( err != 0) {
3451  if( err != ETIMEDOUT) {
3452  lslogging_log_message( "lspmac_moveabs_wait: unexpected error from timedwait %d tv_sec %ld tv_nsec %ld", err, timeout.tv_sec, timeout.tv_nsec);
3453  }
3454  return 1;
3455  }
3456 
3457  //
3458  // wait for the motion to have started
3459  // This will time out if the motion ends before we can read the status back
3460  // hence the added complication of time stamp of the sent packet.
3461 
3462  err = 0;
3463  pthread_mutex_lock( &(mp->mutex));
3464  while( err == 0 && mp->motion_seen == 0)
3465  err = pthread_cond_timedwait( &(mp->cond), &(mp->mutex), &timeout);
3466 
3467  if( err != 0) {
3468  if( err != ETIMEDOUT) {
3469  lslogging_log_message( "lspmac_moveabs_wait: unexpected error from timedwait: %d tv_sec %ld tv_nsec %ld", err, timeout.tv_sec, timeout.tv_nsec);
3470  }
3471  pthread_mutex_unlock( &(mp->mutex));
3472  return 1;
3473  }
3474 
3475  //
3476  // wait for the motion that we know has started to finish
3477  //
3478  err = 0;
3479  while( err == 0 && mp->not_done)
3480  err = pthread_cond_timedwait( &(mp->cond), &(mp->mutex), &timeout);
3481 
3482 
3483  if( err != 0) {
3484  if( err != ETIMEDOUT) {
3485  lslogging_log_message( "lspmac_moveabs_wait: unexpected error from timedwait: %d tv_sec %ld tv_nsec %ld", err, timeout.tv_sec, timeout.tv_nsec);
3486  }
3487  pthread_mutex_unlock( &(mp->mutex));
3488  return 1;
3489  }
3490 
3491  //
3492  // if return code was not 0 then we know we shouldn't wait for not_done flag.
3493  // In this case the motion ended before we read the status registers
3494  //
3495  pthread_mutex_unlock( &(mp->mutex));
3496  return 0;
3497 }
