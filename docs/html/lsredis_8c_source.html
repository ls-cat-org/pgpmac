928 void *lsredis_worker( void *dummy) {
929  static int poll_timeout_ms = -1;
930  static struct pollfd fda[4];
931  static int nfda = 0;
932  static sigset_t our_sigset;
933  int pollrtn;
934  int i;
935 
936 
937  pthread_mutex_lock( &lsredis_mutex);
938  //
939  // block ordinary signal mechanism
940  //
941  sigemptyset( &our_sigset);
942  sigaddset( &our_sigset, SIGUSR1);
943  pthread_sigmask( SIG_BLOCK, &our_sigset, NULL);
944 
945  // Set up fd mechanism
946  //
947  fda[0].fd = signalfd( -1, &our_sigset, SFD_NONBLOCK);
948  if( fda[0].fd == -1) {
949  char *es;
950 
951  es = strerror( errno);
952  lslogging_log_message( "lsredis_worker: Signalfd trouble '%s'", es);
953  }
954  fda[0].events = POLLIN;
955  nfda = 1;
956 
957  lsredis_running = 1;
958 
959  if( redisAsyncCommand( subac, lsredis_subCB, NULL, "PSUBSCRIBE REDIS_KV_CONNECTOR UI* MD2-*") == REDIS_ERR) {
960  lslogging_log_message( "Error sending PSUBSCRIBE command");
961  }
962 
963  redisAsyncCommand( roac, lsredis_keysCB, NULL, "KEYS *");
964  pthread_cond_signal( &lsredis_cond);
965  pthread_mutex_unlock( &lsredis_mutex);
966 
967 
968  while(1) {
969  nfda = 1;
970 
971  pthread_mutex_lock( &lsredis_mutex);
972  if( subfd.fd != -1) {
973  fda[nfda].fd = subfd.fd;
974  fda[nfda].events = subfd.events;
975  fda[nfda].revents = 0;
976  nfda++;
977  }
978 
979  if( rofd.fd != -1) {
980  fda[nfda].fd = rofd.fd;
981  fda[nfda].events = rofd.events;
982  fda[nfda].revents = 0;
983  nfda++;
984  }
985 
986  if( wrfd.fd != -1) {
987  fda[nfda].fd = wrfd.fd;
988  fda[nfda].events = wrfd.events;
989  fda[nfda].revents = 0;
990  nfda++;
991  }
992  pthread_mutex_unlock( &lsredis_mutex);
993 
994  pollrtn = poll( fda, nfda, poll_timeout_ms);
995 
996  if( pollrtn && fda[0].revents) {
997  lsredis_sig_service( &(fda[0]));
998  pollrtn--;
999  }
1000 
1001  for( i=1; i<nfda; i++) {
1002  if( fda[i].revents) {
1003  lsredis_fd_service( &(fda[i]));
1004  }
1005  }
1006  }
1007 }
