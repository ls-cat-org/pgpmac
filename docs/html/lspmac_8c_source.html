1613 void lspmac_next_state() {
1614 
1615 
1616  //
1617  // Connect to the pmac and perhaps initialize it.
1618  // OK, this is slightly more than just the state
1619  // machine logic...
1620  //
1621  if( ls_pmac_state == LS_PMAC_STATE_DETACHED) {
1622  //
1623  // TODO (eventually)
1624  // This ip address wont change in a single PMAC installation
1625  // We'll need to audit the code if we decide to implement
1626  // multiple PMACs so might as well wait til then.
1627  //
1628  lsConnect( "192.6.94.5");
1629 
1630  //
1631  // If the connect was successful we can proceed with the initialization
1632  //
1633  if( ls_pmac_state != LS_PMAC_STATE_DETACHED) {
1634  lspmac_SockFlush();
1635 
1636  //
1637  // Harvest the I and M variables in case we need them
1638  // one day.
1639  //

1649  }

1654  //
1655  if( ls_pmac_state == LS_PMAC_STATE_IDLE && ethCmdOn != ethCmdOff)
1656  ls_pmac_state = LS_PMAC_STATE_SC;
1657

1670

1677 
1678  lspmac_get_status();
1679  }
1680 
1681 
1682 
1683  //

1685  //

1687  case LS_PMAC_STATE_WACK:
1688  case LS_PMAC_STATE_WACK_CC:
1689  case LS_PMAC_STATE_WACK_RR:
1690  case LS_PMAC_STATE_WCR:
1691  case LS_PMAC_STATE_WGB:
1692  case LS_PMAC_STATE_GMR:

1698  //

1700  case LS_PMAC_STATE_CR:
1701  case LS_PMAC_STATE_RR:
1702  case LS_PMAC_STATE_GB:

2008  // hence the added complication of time stamp of the sent packet.
2009  //
2010 
2011  return_code=0;
2012 
2013  pthread_mutex_lock( &(mp->mutex));
2014  while( mp->motion_seen == 0 && return_code == 0)
2015  return_code = pthread_cond_timedwait( &(mp->cond), &(mp->mutex), &wt);
2016 
2017  if( return_code == 0) {
2018  //
2019  // wait for the motion that we know has started to finish
2020  //
2021  while( mp->not_done)
2022  pthread_cond_wait( &(mp->cond), &(mp->mutex));
2023
