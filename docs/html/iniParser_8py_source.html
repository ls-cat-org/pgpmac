25 # We assume the sections and options are case insensitive and that,
26 # although nested sections are implied by the format used by the md2,
27 # that the nesting has no practical importance.
28 #
29 # The current version is for READING the files.

31 # TODO: add writing. We'll need to keep track of the preferred case

36 # this is not necessary as we can just spit stuff out all lower case

53  for l in self.f.readlines():
54  sl = l.strip()
55  if len(sl) > 0:
56  if sl[0] == ";":
57  continue
58 
59  if sl[0] == "[" and sl.find("]") > 1:
60  self.sd[current_section] = current_dict
61  current_dict = {}
62  current_section = (sl[1:sl.find("]")]).lower()
63 
64  else:
65  if sl.find(";") > 0:

67  else:
68  s = sl
69

72  if len(slist) == 2:
73  k = (slist[0].strip()).lower()
74  v = slist[1].strip()
75  current_dict[k] = v
